function formatName(name = "") {
  return console.log(
    name
      .split(" ")
      .map(
        (char) => char.charAt(0, 1).toUpperCase() + char.slice(1).toLowerCase()
      )
      .join(" ")
  );
}

formatName("eGGi aUnur rOfiq"); // Output: "Eggi Aunur Rofiq"
formatName("jOhn DOE"); // Output: "John Doe"
